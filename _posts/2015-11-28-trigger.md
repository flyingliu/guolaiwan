trigger相对于click等普通的预定义事件，主要有两点不一样。

1. trigger可以接受多个参数。

2. trigger可以自定义事件

下面是实例代码

    (function($){
      var btn = $("#btn");
      btn.bind('clicka',function(e,a,b,c){
        console.log(a,b,c)
      });
      btn.click(function(){
        btn.trigger('clicka',['hello','world','flying'])  // hello world flying    
      })
      btn.trigger('clicka',['a','b','c']); // a,b,c  
    })(jQuery);
